Create a function that counts the number of times a particular letter shows up in the word search.


### Examples ###
    letterCounter([
      ["D", "E", "Y", "H", "A", "D"],
      ["C", "B", "Z", "Y", "J", "K"],
      ["D", "B", "C", "A", "M", "N"],
      ["F", "G", "G", "R", "S", "R"],
      ["V", "X", "H", "A", "S", "S"]
    ], "D") ➞ 3

    // "D" shows up 3 times: twice in first row, once in third row.

    letterCounter([
      ["D", "E", "Y", "H", "A", "D"],
      ["C", "B", "Z", "Y", "J", "K"],
      ["D", "B", "C", "A", "M", "N"],
      ["F", "G", "G", "R", "S", "R"],
      ["V", "X", "H", "A", "S", "S"]
    ], "H") ➞ 2
